# Collapse
## Installation
Using the react components:

    import Collapse from '@vividseats/vivid-design-patterns/dist/molecules/Collapse';

Using the scss files:

    @import '@vividseats/vivid-design-patterns/src/styles/_collapse.scss';

## As a controlled component
<Playground>
    <StatefulComponent initialState={{open: false}}>
    {
        ({state, setState}) => {
            const onOpenChange = (open) => setState({open:open});
            const { open } = state;
            return (
                    <Collapse onOpenChange={onOpenChange} open={open} title={"Controlled"}>
                        <LinkGroup type="muted">
                            <Link state="muted" weight="regular" href="/about.html">
