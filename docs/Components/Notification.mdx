## Basic Usage

<Playground>
    <StatefulComponent initialState={{open: true}}>
            {
        ({state, setState}) => {
            const { open } = state;

            const handleClose = () => {
                setState({open: false})
                setTimeout(() => setState({open: true}), 2000)
            }

            return (
                <Notification type="toast" isOpen={open} onClickClose={handleClose} style={{backgroundColor: 'lightblue'}}>
                    <Headline importance={3}>Vivid Seats </Headline>
                    <Headline importance={3}>Buy In-App,</Headline>
                    <Headline importance={3}>Earn Credit Back</Headline>
                </Notification>
            )
        }
    }
    </StatefulComponent>
